### <a name="violation"></a><span data-ttu-id="7e5f9-132">Violation</span><span class="sxs-lookup"><span data-stu-id="7e5f9-132">Violation</span></span>

```csharp
using Newtonsoft.Json;

public class ExampleClass
{
    public JsonSerializerSettings Settings { get; }

    public ExampleClass()
    {
        Settings = new JsonSerializerSettings();
        Settings.TypeNameHandling = TypeNameHandling.All;    // CA2326 violation.
    }
}
```

```vb
Imports Newtonsoft.Json

Public Class ExampleClass
    Public ReadOnly Property Settings() As JsonSerializerSettings

    Public Sub New()

        Settings.TypeNameHandling = TypeNameHandling.All    ' CA2326 violation.
    End Sub
End Class
```

### <a name="solution"></a><span data-ttu-id="7e5f9-133">Solution</span><span class="sxs-lookup"><span data-stu-id="7e5f9-133">Solution</span></span>

```csharp
using Newtonsoft.Json;

public class ExampleClass
{
    public JsonSerializerSettings Settings { get; }

    }
}
```
