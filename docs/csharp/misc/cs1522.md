```csharp

{  
   public static void Main()  
   {  
      int i = 6;

      {  
         // add something to the switch block, for example:  
         /*

            Console.WriteLine("5");  
            return;  
         default:  
            Console.WriteLine("not 5");  
            return;  
         */  
      }  
   }  
}  
```
