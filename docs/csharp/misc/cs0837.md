## <a name="example"></a><span data-ttu-id="73ccd-110">Exemple</span><span class="sxs-lookup"><span data-stu-id="73ccd-110">Example</span></span>

    public delegate void Del();

    {
        static int Main()
        {
            bool b1 = (() => { }) is Del;   // CS0837

        }
    }
}
