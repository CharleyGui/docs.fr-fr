   public class A { }  
}  
```  
  
 <span data-ttu-id="56c98-109">Compilez ensuite ce fichier :</span><span class="sxs-lookup"><span data-stu-id="56c98-109">Then, compile this file:</span></span>

{  
   public class A { }  
}

{  
   public static void Main()
   {  
      Console.WriteLine(typeof(Util.A)); // CS0435  
   }  
}  
```
