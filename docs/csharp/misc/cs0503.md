# <a name="compiler-error-cs0503"></a><span data-ttu-id="23096-102">Erreur du compilateur CS0503</span><span class="sxs-lookup"><span data-stu-id="23096-102">Compiler Error CS0503</span></span>

      abstract virtual public void f();   // CS0503
