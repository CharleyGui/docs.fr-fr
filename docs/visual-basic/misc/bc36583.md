```vb  
<Extension()> _  
Public Sub ABC(ByVal X As Integer, Optional ByVal Y As Byte = 0, _  
               Optional ByVal Z As Byte = 0)  
End Sub

' Calling extension method ABC.  
Dim number As Integer  
' Not valid.
```  
