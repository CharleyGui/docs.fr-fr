open System
open System.Data
open System.Data.Linq
open Microsoft.FSharp.Data.TypeProviders

query {
    for row in db.Table2 do
    where (row.TestData1.HasValue && row.TestData1.Value > 2)

} |> Seq.iter (fun row -> printfn "%d %s" row.TestData1.Value row.Name)
